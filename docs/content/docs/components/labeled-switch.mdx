## Usage

```tsx
import { LabeledSwitch } from "@/components/ui/labeled-switch";
```

```tsx
const [selected, setSelected] = React.useState<boolean>(false);

return (
  <LabeledSwitch
    firstLabel='False'
    secondLabel='True'
    selected={selected}
    onToggle={setSelected}
  />
);
```
