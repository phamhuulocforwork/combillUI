## Usage

```tsx
import {
  AnimatedInput,
  AnimatedLabel,
  AnimatedLabelInput, // If you want to use the component as a whole
} from "@/components/ui/animated-label-input";
```

### Default

```tsx
<AnimatedLabelInput label='Animated Label' />
```

### Customize

```tsx
<AnimatedInput id="animated-label-input-demo" />
<AnimatedLabel htmlFor="animated-label-input-demo">Animated Label</AnimatedLabel>
```
