## Usage

```tsx
import {
  AnimatedTooltip,
  AnimatedTooltipContent,
  AnimatedTooltipProvider,
  AnimatedTooltipTrigger,
} from "@/components/ui/animated-tooltip";
```

```tsx
<AnimatedTooltipProvider>
  <AnimatedTooltip>
    <AnimatedTooltipTrigger>Hover</AnimatedTooltipTrigger>
    <AnimatedTooltipContent className='bg-black'>
      <p>Add to library</p>
    </AnimatedTooltipContent>
  </AnimatedTooltip>
</AnimatedTooltipProvider>
```
