## Usage

```tsx
import { useClickOutside } from "@/hooks/use-click-outside";
```

```tsx
const [isOpen, setIsOpen] = React.useState(false);

// Ref to the element you want to detect clicks outside of
const ref = useClickOutside(() => {
  setIsOpen(false);
});
```
