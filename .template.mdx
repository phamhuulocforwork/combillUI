## Usage

```tsx
import {
  ComponentName,
  ComponentSubPart1,
  ComponentSubPart2,
} from "@/components/ui/component-name";
```

### Default

```tsx
<ComponentName />
```

### Variants

#### Variant 1

```tsx
<ComponentName variant='variant1' />
```

#### Variant 2

```tsx
<ComponentName variant='variant2' />
```

### Customization

```tsx
<ComponentSubPart1 className="custom-class" />
<ComponentSubPart2>Custom content</ComponentSubPart2>
```
